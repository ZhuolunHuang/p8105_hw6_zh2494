``` r
bootstrap_df =
  simple_df %>% 
  modelr::bootstrap(n = 5000) %>% 
  mutate(
    models = map(.x = strap, ~lm(tmax ~ tmin, data = .x)),
    result1 = map(models, broom::glance),
    result2 = map(models, broom::tidy)
  ) %>% 
  select(-strap, -models) %>% 
  unnest(result1, result2) %>% 
  select(.id, term, estimate, r.squared) %>% 
  group_by(.id) %>% 
  mutate(
```
